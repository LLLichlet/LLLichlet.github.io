\section{Quotient Space}

\begin{definition}
Let $V$ be a vector space over a field $F$ and $U \subseteq V$ a subspace. For $v \in V$, the \textbf{coset} of $v$ modulo $U$ is
\[
v + U = \{\, v + u \mid u \in U \,\}.
\]
The \textbf{quotient space} $V / U$ is the set of all cosets
\[
V / U = \{\, v + U \mid v \in V \,\},
\]
with addition and scalar multiplication defined by
\[
(v + U) + (w + U) = (v + w) + U, \qquad
\lambda(v + U) = \lambda v + U,
\]
for $v, w \in V$ and $\lambda \in F$.
\end{definition}

\begin{definition}
The \textbf{quotient map} $\pi \colon V \to V / U$ is defined by
\[
\pi(v) = v + U.
\]
\end{definition}

\begin{proposition}
$\pi$ is a linear map, $\ker \pi = U$, and $\operatorname{im} \pi = V / U$.
\end{proposition}

\begin{proof}
Left as an exercise.
\end{proof}

\begin{proposition}
$V / U$ is a vector space over $F$.
\end{proposition}

\begin{proof}
Left as an exercise.
\end{proof}

\begin{proposition}
For any linear map $f \colon V \to W$,
\[
V / \ker f \cong \operatorname{im} f.
\]
\end{proposition}

\begin{proof}
Define $\Phi \colon V / \ker f \to \operatorname{im} f$ by $\Phi(v + \ker f) = f(v)$.

Well-defined: if $v + \ker f = w + \ker f$, then $v - w \in \ker f$, so $f(v) - f(w) = f(v - w) = 0$, hence $f(v) = f(w)$.

$\Phi$ is linear: $\Phi((v + \ker f) + (w + \ker f)) = \Phi((v + w) + \ker f) = f(v + w) = f(v) + f(w) = \Phi(v + \ker f) + \Phi(w + \ker f)$, and similarly for scalar multiplication.

Injective: if $\Phi(v + \ker f) = 0$, then $f(v) = 0$, so $v \in \ker f$, hence $v + \ker f = 0 + \ker f$.

Surjective: for any $f(v) \in \operatorname{im} f$, we have $\Phi(v + \ker f) = f(v)$.
\end{proof}
