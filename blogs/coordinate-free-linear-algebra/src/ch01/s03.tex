\section{Linear Maps}

\begin{definition}
Let $V$, $W$ be vector spaces over a field $F$. A map $f \colon V \to W$ is \textbf{linear} if
\begin{itemize}
    \item $f(u + v) = f(u) + f(v)$ for all $u, v \in V$,
    \item $f(\lambda v) = \lambda f(v)$ for all $\lambda \in F$, $v \in V$.
\end{itemize}
\end{definition}

\begin{definition}
The set of all linear maps from $V$ to $W$ is denoted
\[
\operatorname{Hom}(V, W) = \{\, f \colon V \to W \mid f \text{ is linear} \,\}.
\end{definition}

The projection maps
\[
\pi_{1} \colon V \times W \to V,\; (v, w) \mapsto v,
\qquad
\pi_{2} \colon V \times W \to W,\; (v, w) \mapsto w,
\]
are linear. More generally, for any family $\{V_{i}\}_{i \in I}$, the projections $\pi_{j} \colon \prod_{i \in I} V_{i} \to V_{j}$ defined by $(v_{i})_{i \in I} \mapsto v_{j}$ are linear. (Exercise.)

\begin{definition}
For a linear map $f \colon V \to W$, the \textbf{kernel} and \textbf{image} are
\[
\ker f = \{\, v \in V \mid f(v) = 0 \,\}, \qquad
\operatorname{im} f = \{\, f(v) \mid v \in V \,\}.
\]
\end{definition}

\begin{proposition}
$\ker f$ is a subspace of $V$, and $\operatorname{im} f$ is a subspace of $W$.
\end{proposition}

\begin{proof}
Left as an exercise.
\end{proof}

\begin{proposition}
A linear map $f \colon V \to W$ is injective if and only if $\ker f = \{0\}$.
\end{proposition}

\begin{proof}
Left as an exercise.
\end{proof}

\begin{definition}
A linear map $f \colon V \to W$ is an \textbf{isomorphism} if it is bijective. Two vector spaces $V$, $W$ are \textbf{isomorphic}, denoted $V \cong W$, if there exists an isomorphism $V \to W$.
\end{definition}
