\section{Isomorphism Theorems}

\begin{proposition}
Let $U, W \subseteq V$ be subspaces. Then
\[
(U + W) / W \cong U / (U \cap W).
\]
\end{proposition}

\begin{proof}
Left as an exercise. (Hint: define a suitable linear map on $U$ and apply Proposition 1.4.5.)
\end{proof}

\begin{proposition}
Let $U \subseteq W \subseteq V$ be subspaces. Then
\[
(V / U) \big/ (W / U) \cong V / W.
\]
\end{proposition}

\begin{proof}
Left as an exercise. (Hint: define a suitable linear map on $V/U$ and apply Proposition 1.4.5.)
\end{proof}

\begin{proposition}
For any subspaces $U, W \subseteq V$,
\[
\dim(U + W) + \dim(U \cap W) = \dim U + \dim W.
\]
\end{proposition}

\begin{proof}
Left as an exercise. (Hint: apply Proposition 1.6.1 and Proposition 1.5.3.)
\end{proof}
